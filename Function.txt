#include <iostream>
using namespace std;

int factorial(int n)
{
      if(n == 0)
           return 1;
      int sum = 1;
      for(int i=1; i<=n; i++)
      {
          sum = sum * i;

      }
      return sum;



}
int main()
{
   int k;
   cin>>k;

   cout<<factorial(k);

}
