int main()
{
    
    int a,b;
    cin>>a>>b;


    int option;
    cin>>option;

    if(option == 1)
        cout<<a+b;

    else if(option == 2)
        cout<<a-b;

    else if(option == 3)
        cout<<a*b;

    else if(option == 3)
        cout<<a/b;

    else
        cout<<"Invalid Input"<<endl;


}
